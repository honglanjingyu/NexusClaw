## 如何添加自定义工具

在当前的 Agent 架构中，添加新工具非常简单。以下是完整的添加步骤：

### 方法一：在已有工具文件中添加（推荐用于简单工具）

#### 步骤1：在 `app/action/tools/` 下创建或编辑工具文件

例如，创建一个计算器工具 `calculator_tool.py`：

```python
# app/action/tools/calculator_tool.py
"""计算器工具 - 执行数学计算"""

import math
from typing import Optional
from loguru import logger

async def calculate(
    expression: str,
    precision: Optional[int] = 2
) -> str:
    """
    执行数学计算
    
    Args:
        expression: 数学表达式，如 "1+2*3" 或 "sqrt(16)"
        precision: 结果保留小数位数，默认2位
    
    Returns:
        str: 计算结果
    """
    logger.info(f"计算器: expression='{expression}', precision={precision}")
    
    # 允许的函数
    safe_functions = {
        'abs': abs,
        'round': round,
        'min': min,
        'max': max,
        'sum': sum,
        'sqrt': math.sqrt,
        'pow': math.pow,
        'sin': math.sin,
        'cos': math.cos,
        'tan': math.tan,
        'log': math.log,
        'log10': math.log10,
        'pi': math.pi,
        'e': math.e
    }
    
    try:
        # 使用 eval 但限制可用函数
        # 注意：生产环境建议使用更安全的解析器如 numexpr
        result = eval(expression, {"__builtins__": {}}, safe_functions)
        
        # 格式化结果
        if isinstance(result, (int, float)):
            if precision is not None:
                result = round(result, precision)
            return f"计算结果: {expression} = {result}"
        else:
            return f"计算结果: {expression} = {result}"
            
    except ZeroDivisionError:
        return "错误: 除数不能为零"
    except Exception as e:
        logger.error(f"计算失败: {e}")
        return f"计算错误: {str(e)}"


async def get_weather(city: str) -> str:
    """
    获取天气信息（示例）
    
    Args:
        city: 城市名称
    
    Returns:
        str: 天气信息
    """
    logger.info(f"查询天气: city='{city}'")
    
    # 这里可以调用真实的天气 API
    # 示例返回模拟数据
    weather_data = {
        "北京": "晴天，25°C，湿度45%",
        "上海": "多云，28°C，湿度60%",
        "广州": "阵雨，30°C，湿度80%",
        "深圳": "晴天，32°C，湿度55%",
    }
    
    result = weather_data.get(city, f"未找到{city}的天气信息")
    return f"{city}天气: {result}"
```

#### 步骤2：在 `app/action/tools/__init__.py` 中导出新工具

```python
# app/action/tools/__init__.py
"""内置行动工具"""

from .time_tool import get_current_time
from .knowledge_tool import (
    search_knowledge,
    search_knowledge_with_filter,
    get_knowledge_stats,
    add_to_knowledge
)
from .calculator_tool import calculate, get_weather  # 添加新工具

__all__ = [
    "get_current_time",
    "search_knowledge",
    "search_knowledge_with_filter",
    "get_knowledge_stats",
    "add_to_knowledge",
    "calculate",      # 新工具
    "get_weather",    # 新工具
]
```

#### 步骤3：在 `app/action/action_manager.py` 中注册新工具

修改 `register_builtin_tools` 方法：

```python
# app/action/action_manager.py

def register_builtin_tools(self) -> None:
    """注册内置工具"""
    from app.action.tools import (
        get_current_time,
        search_knowledge,
        search_knowledge_with_filter,
        get_knowledge_stats,
        add_to_knowledge,
        calculate,        # 添加新工具
        get_weather,      # 添加新工具
    )

    self.register_tools([
        get_current_time,
        search_knowledge,
        search_knowledge_with_filter,
        get_knowledge_stats,
        add_to_knowledge,
        calculate,        # 注册新工具
        get_weather,      # 注册新工具
    ])
```

### 工具模板

新工具的标准模板：

```python
# app/action/tools/your_tool.py
"""
工具功能描述
"""

from typing import Optional, Dict, Any, List
from loguru import logger

# 同步函数示例
def sync_function(param1: str, param2: int = 10) -> str:
    """
    函数描述
    
    Args:
        param1: 参数1说明
        param2: 参数2说明，默认值10
    
    Returns:
        str: 返回值说明
    """
    # 实现逻辑
    result = f"处理结果: {param1}, {param2}"
    return result


# 异步函数示例（推荐）
async def async_function(
    query: str,
    limit: int = 5,
    filters: Optional[Dict[str, Any]] = None
) -> str:
    """
    异步函数描述
    
    Args:
        query: 查询内容
        limit: 返回数量限制
        filters: 过滤条件
    
    Returns:
        str: 格式化后的结果
    """
    logger.info(f"调用异步工具: query='{query}', limit={limit}")
    
    try:
        # 你的业务逻辑
        # 可以调用 API、查询数据库等
        
        result = f"查询结果: {query}"
        return result
        
    except Exception as e:
        logger.error(f"工具执行失败: {e}")
        return f"错误: {str(e)}"
```

### 测试新工具

直接测试：
```python
# 测试脚本
async def test_new_tool():
    from app.action import get_action_manager
    
    manager = get_action_manager()
    manager.register_builtin_tools()
    
    # 测试计算器
    result = await manager.execute_tool_call(
        tool_name="calculate",
        tool_input={"expression": "2+3*4", "precision": 2}
    )
    print(result)  # 计算结果: 2+3*4 = 14
    
    # 测试天气
    result = await manager.execute_tool_call(
        tool_name="get_weather",
        tool_input={"city": "北京"}
    )
    print(result)  # 北京天气: 晴天，25°C，湿度45%
```
